Willer Framework

Willer is a PHP framework, it was created based on ideas coming from other frameworks like Codeigniter(php) and Django(python).

Prerequisites & Dependencies

  • PHP >= 5.6
  • Nginx >= 1.2

Download & Install

Make direct git clone the GitHub project page on your folder localhost

Usage

Example with transactions, controller Application\Example\Controller\Home

<?php
namespace Application\Example\Controller {

    use \Exception as Exception;
    use \Core\Controller;
    use \Core\DAO\Transaction;
    use \Application\Log\Model\Log;

    class Home extends Controller {
        public function __construct($request_method = null) {

            parent::__construct($request_method);

            $this->transaction_mysql = new Transaction(DB_MYSQL);
        }

        public function index($url_fragment) {

            $log_error = new Log\Error($this->transaction_mysql);
            $log_errortype = new Log\ErrorType($this->transaction_mysql);
            $log_user = new Log\User($this->transaction_mysql);
            $log_register = new Log\Register($this->transaction_mysql);

            try {

                $this->transaction_mysql->beginTransaction();

                $log_errortype->save([
                    "name" => "error_type_test"]);

                /* or ...
                // $log_errortype->name = "error_test";
                // $log_errortype->save();
                */

                $log_error->save([
                    "name" => "error_test",
                    "tipe_id" => $log_errortype,
                    "describe" => "test describe"]);

                /* or ...
                // $log_error->name = "error_test";
                // $log_error->describe = "test describe";
                // $log_error->save();
                */

                $log_user->save([
                    "name" => "user_test"
                    "publickey" => "123456"]);

                $log_register->save([
                    "user_id" => $log_user,
                    "error_id" => $log_error]);

                $log_register_list = $log_register
                    ->where(["register.id" => [1,]])
                    ->orderBy(["id" => "asc"])
                    ->limit(1,5)
                    ->execute(["join" => "left"]);

                /* view last query
                print_r($log_register->lastQuery());
                */

                foreach ($log_register_list as $i => $obj_register) {
                    $obj_register->user_id->name = "other user name";
                    $obj_register->user_id->save();

                    $obj_register->error_id->describe = "other error description";
                    $obj_register->error_id->save();

                    $obj_register->error_id->type_id->name = "other error_type name";
                    $obj_register->error_id->type_id->save();
                }

                $this->transaction_mysql->commit();

            } catch (Exception $error) {

                $this->transaction_mysql->rollBack();

                throw new Exception($error);
            }
        }
    }
}
